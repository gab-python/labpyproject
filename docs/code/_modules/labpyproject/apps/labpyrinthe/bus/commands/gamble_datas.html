        """
        Retourne un filtre de liste (sous forme de dict)
        """
        if filter_type == cls.FILTER_LT:
            return {"filtre": cls.filter_property_lt, "attr": attr, "value": val}

        """
        Pour filtrer une liste de GambleDataObject en slectionnat ceux dont l'attribut 
        prop est infrieur  val
        """
        if hasattr(gdo, attr):
            if gdo[attr] < val:

        return False

[docs]    def get_GDObj_list_by_dir(self, direct, filterdict=None):
        """
        Retourne la liste des GambleDataObject associe  une direction
        """
        if direct in LabHelper.LIST_DIRECTIONS:
            liste = self._innerdict[direct]
            if filterdict == None:
                return liste
            else:
                fct_filtre = filterdict["filtre"]
                attr = filterdict["attr"]
                val = filterdict["value"]
                rlist = list()
                for gdo in liste:
                    if fct_filtre(gdo, attr, val):
                        rlist.append(gdo)
                return rlist
        return None
